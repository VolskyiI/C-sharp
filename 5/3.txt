using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace _3
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] a = new int[50];
            long suma = 0;
            Random ran = new Random();
            for (int i = 0; i < a.Length; i++)
            {
                a[i] = ran.Next(-20,20);
                Console.WriteLine(a[i]);
                if (a[i] >= 0)
                {
                    suma += a[i];
                }
            }
            Console.WriteLine($"Сумма додатніх елементів масива = {suma}");
            Console.ReadLine();
        }
    }
